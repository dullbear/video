190 		timeFormat: function(time) {
191 			//格式化时间
192 			var hour = parseInt(time / 3600),
193 				minute = parseInt(time / 60),
194 				second = parseInt(time % 60),
195 				html = '';
196 			if (hour < 1) {
197 				html += minute >= 10 ? minute + ':' : '0' + minute + ':';
198 				html += second >= 10 ? second : '0' + second;
199 			} else {
200 				html += hour >= 10 ? hour + ':' : '0' + hour + ':';
201 				html += minute >= 10 ? minute + ':' : '0' + minute + ':';
202 				html += second >= 10 ? second : '0' + second;
203 			}
204 
205 			return html;

238 					_self.$bufferProg.animate({
239 						'width': bufferWid
240 					});
241 
242 					//缓冲完成
243 					if (end == _self.alltime) {
244 						clearInterval(time01);
245 					};
246

248 
249 				//加载
250 				if (this.$video.readyState == 2) {
